🔹 1. LIST (ArrayList / LinkedList)
✅ 1. Average of marks
import java.util.*;

public class AvgMarks {
    public static void main(String[] args) {
        ArrayList<Integer> marks = new ArrayList<>(Arrays.asList(50, 60, 70));

        int sum = 0;
        for (int m : marks) sum += m;

        System.out.println("Average = " + (sum / marks.size()));
    }
}

✅ 3. Sort using Comparable
class Student implements Comparable<Student> {
    int marks;
    Student(int m){ marks = m; }

    public int compareTo(Student s) {
        return this.marks - s.marks; // ascending
    }
}

✅ 4. Highest frequency element
Map<Integer,Integer> map = new HashMap<>();
int arr[] = {1,2,2,3};

for(int n: arr)
    map.put(n, map.getOrDefault(n,0)+1);

int maxKey = -1, maxVal = 0;
for(var e: map.entrySet()){
    if(e.getValue() > maxVal){
        maxVal = e.getValue();
        maxKey = e.getKey();
    }
}
System.out.println(maxKey);

✅ Anagrams grouping
String[] words = {"eat","tea","tan"};

Map<String,List<String>> map = new HashMap<>();

for(String w: words){
    char ch[] = w.toCharArray();
    Arrays.sort(ch);
    String key = new String(ch);

    map.computeIfAbsent(key,k->new ArrayList<>()).add(w);
}
System.out.println(map);
